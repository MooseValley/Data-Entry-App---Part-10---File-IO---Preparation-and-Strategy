   private void exitButtonHandler(ActionEvent event)
    {
        try (Formatter outFile = new Formatter ("courses.txt") )
        {
            for (Course c : CourseAddFXMLController.coursesArrayList)
            {
                outFile.format (c.getCode() + "\n" +
                                c.getName() + "\n");
            }
        }
        catch (Exception err)
        {
            JOptionPane.showMessageDialog (null, "Error: courses file write failed.");
        }

        StudentAdminApp.exitApplication (null);

        System.exit (0); // Exit app
    }
